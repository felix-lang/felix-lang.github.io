1.2 List constructors.

  //Check List::_ctor_list
  begin
    fun str (x:list[int]) =>
      match x with
      | #Empty => "()"
      | Cons (h, t) => "(" + (str h) + ", " + (str t) + ")"
      endmatch
    ;
    print$ str$ list[int](); endl;
    print$ str$ list(4); endl;
    print$ str$ list(4, 5); endl;
    print$ str$ list(4, 5, 6); endl;
    print$ str$ list(4, 5, 6, 7); endl;
  end
